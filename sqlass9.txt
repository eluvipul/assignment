Queries:

1.-- Create influencers table
CREATE TABLE influencers (
    influencer_id INT PRIMARY KEY,
    influencer_name VARCHAR(50),
    city VARCHAR(50)
);

-- Create brands table
CREATE TABLE brands (
    brand_id INT PRIMARY KEY,
    brand_name VARCHAR(50),
    city VARCHAR(50)
);

-- Insert sample rows
INSERT INTO brands (brand_id, brand_name, city)
VALUES
(1, 'TechCorp', 'New York'),
(2, 'StyleHub', 'Miami'),
(3, 'Foodies', 'Chicago');

-- FULL OUTER JOIN on city
SELECT 
    i.influencer_name,
    b.brand_name,
    i.city AS influencer_city,
    b.city AS brand_city
FROM influencers i
FULL OUTER JOIN brands b
ON i.city = b.city
ORDER BY i.city, b.city;

2.SELECT 
    child.playlist_name AS child_playlist,
    parent.playlist_name AS parent_playlist
FROM playlists child
LEFT JOIN playlists parent
    ON child.parent_playlist_id = parent.id
ORDER BY parent.playlist_name, child.playlist_name;

4.-- Show each employee with their manager
SELECT 
    e.name AS employee_name,
    m.name AS manager_name
FROM employees e
LEFT JOIN employees m
    ON e.manager_id = m.id
ORDER BY manager_name, employee_name;

    u1.user_name AS user1,
    u2.user_name AS user2,
    u1.city
FROM users u1
INNER JOIN users u2
    ON u1.city = u2.city
   AND u1.id <> u2.id
ORDER BY u1.city, u1.user_name, u2.user_name;
