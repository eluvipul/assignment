import pandas as pd
from sqlalchemy import create_engine

# Step 1: Connect to your SQLite database
engine = create_engine("sqlite:///my_database.db")  # replace with your actual DB file path

# Step 2: Define and run the SQL query
query = """
SELECT user_id, COUNT(order_id) AS total_orders
FROM orders
GROUP BY user_id
HAVING COUNT(order_id) > 5;
"""

# Step 3: Load results into a pandas DataFrame
df = pd.read_sql(query, engine)

# Step 4: Display the DataFrame
print(df.head())
