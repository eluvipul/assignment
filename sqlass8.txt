5.CREATE TABLE CustomerSegments (
    segment_id INT PRIMARY KEY,
    segment_name VARCHAR(50) NOT NULL
);

ALTER TABLE Users
ADD segment_id INT,
ADD CONSTRAINT fk_segment FOREIGN KEY (segment_id)
REFERENCES CustomerSegments(segment_id);

SELECT u.username, cs.segment_name, SUM(o.order_amount) AS total_order_amount
FROM Users u
INNER JOIN CustomerSegments cs ON u.segment_id = cs.segment_id
INNER JOIN Orders o ON u.user_id = o.user_id
GROUP BY u.username, cs.segment_name
ORDER BY cs.segment_name, total_order_amount DESC;
