import pandas as pd
from sqlalchemy import create_engine

# Step 1: Read the CSV file
df = pd.read_csv("mi_matches.csv")

# Step 2: Filter matches where Mumbai Indians won
mi_wins_df = df[df['winner'] == 'Mumbai Indians']

# Step 3: Connect to SQL database (replace with your actual connection string)
# Example for PostgreSQL:
engine = create_engine("postgresql://username:password@localhost:5432/your_database")

# Step 4: Insert filtered rows into a new table called mi_wins
mi_wins_df.to_sql("mi_wins", engine, if_exists="replace", index=False)

print("Inserted Mumbai Indians wins into mi_wins table successfully!")
